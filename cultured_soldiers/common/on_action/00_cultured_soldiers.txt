on_legion_raised = {
	# what: on the end of the day after a legion is created
  	# scope: legion
	effect = {
		random_legion_unit = {
				unit_location.governorship = { # Do I need to scope to a unit?
	        	save_scope_as = this_gov
					set_variable = {
						name = governorship_pops_of_country_culture
						value = 0
					}
					set_variable = {
						name = governorship_pops_of_other_cultures
						value = 0
					}

					every_governorship_state = {
						every_state_province = {
							every_pops_in_province = {
								limit = { # Get pops of country culture group
									culture_group = root.owner.culture_group
								}
								scope:this_gov = { # Add to total of pops of country's culture group in the whole state
									change_variable = {
										name = governorship_pops_of_country_culture
										add = 1
									}
								}
							}
							every_pops_in_province = {
								limit = { # Get pops of other culture groups
									NOT = { culture_group = root.owner.culture_group }
								}
								scope:this_gov = { # Add to total of pops of other cultures group in the whole state
									change_variable = {
										name = governorship_pops_of_other_cultures
										add = 1
									}
								}
							}
						}
					}

					set_variable = { # Get percentage of country culture group pops here
						name = governorship_country_culture_percentage
						value = VAR:governorship_pops_of_country_culture
					}
					set_variable = {
						name = governorship_country_pops_total
						value = VAR:governorship_pops_of_country_culture
					}
					change_variable = {
						name = governorship_country_pops_total
						add = VAR:governorship_pops_of_other_cultures
					}
					change_variable = {
						name = governorship_country_culture_percentage
						divide = VAR:governorship_pops_of_other_cultures
					}
					change_variable = {
						name = governorship_country_culture_percentage
						multiply = 100
					}
					clamp_variable = {
						name = governorship_country_culture_percentage
						max = 100
						min = 0
					}
				}
		}
	}
}

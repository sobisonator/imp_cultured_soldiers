on_legion_raised = {
	# what: on the end of the day after a legion is created
  	# scope: legion
	effect = {
		random_legion_unit = {
				unit_location.governorship = { # Do I need to scope to a unit?
	        	save_scope_as = this_gov
					set_variable = {
						name = governorship_accepted_pops
						value = 0
					}
					set_variable = {
						name = governorship_non_accepted_pops
						value = 0
					}

					every_governorship_state = {
						every_state_province = {
							every_pops_in_province = {
							  save_scope_as = this_pop
							  if = {
							    limit = {
							      root.owner = {
							        any_country_culture = {
							          culture = scope:this_pop.culture
							          OR = {
							            has_pop_type_right = nobles
							            has_country_culture_modifier = officer_right
							            has_country_culture_modifier = honor_guard
							          }
							        }
							      }
							    }
							    scope:this_gov = {
							      change_variable = {
							        name = governorship_accepted_pops
							        add = 1
							      }
							    }
							  }
							}
							every_pops_in_province = {
							  save_scope_as = this_pop
							  if = {
							    limit = {
							      root.owner = {
							        any_country_culture = {
							          culture = scope:this_pop.culture
							          NOR = {
							            has_pop_type_right = nobles
							            has_country_culture_modifier = officer_right
							            has_country_culture_modifier = honor_guard
							          }
							        }
							      }
							    }
							    scope:this_gov = {
							      change_variable = {
							        name = governorship_non_accepted_pops
							        add = 1
							      }
							    }
							  }
							}
						}
					}

					set_variable = { # Get percentage of country culture group pops here
						name = governorship_accepted_culture_percentage
						value = VAR:governorship_accepted_pops
					}
					set_variable = {
						name = governorship_country_pops_total
						value = VAR:governorship_accepted_pops
					}
					change_variable = {
						name = governorship_country_pops_total
						add = VAR:governorship_non_accepted_pops
					}
					change_variable = {
						name = governorship_accepted_culture_percentage
						divide = VAR:governorship_non_accepted_pops
					}
					change_variable = {
						name = governorship_accepted_culture_percentage
						multiply = 100
					}
					clamp_variable = {
						name = governorship_accepted_culture_percentage
						max = 100
						min = 0
					}
				}
		}
	}
}
